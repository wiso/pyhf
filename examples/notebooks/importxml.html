def assemble(pdf, **kwargs):
    pars = pdf.config.suggested_init()
    for k, v in kwargs.items():
        pars[pdf.config.par_slice(k)] = v
    return pars

def disassemble(pdf, pars):
    return {
        k: pars[pdf.config.par_slice(k)] for k in pdf.config.par_map
    }
